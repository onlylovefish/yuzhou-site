字符串相加

可以理解，小学竖式计算

/**
 * @param {string} num1
 * @param {string} num2
 * @return {string}
 */
var addStrings = function(num1, num2) {
    let len1=num1.length,len2=num2.length

    let i=len1-1,j=len2-1

    let arr=[]

    let add=0
    while(i>=0||j>=0||add!=0){
        const x=i>=0?num1.charAt(i)-'0':0
        const y=j>=0?num2.charAt(j)-'0':0

        const sum=x+y+add

        add=Math.floor(sum/10)
        arr.push(sum%10)
        i--
        j--
        console.log(arr)
    }

    return arr.reverse().join('')
};
